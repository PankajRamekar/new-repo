public class{
  public static void main(String[]args){
       checkPrime(5);
		System.out.println("Program for Prime No");
  }
	public static void checkPrime(int num){
		int i=2;
		while(i<=num){
			if(num%i==0){
			    break;
			}i++;
			if(num==i){
				System.out.println("number is prime");
			}	
		}
	}

}
